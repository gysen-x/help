const fs = require('fs');

fs.readFileSync('./index.txt', 'utf-8') - прочитать содержимое файла

fs.writeFileSync('./index.txt', result) - создать файл index.txt и записать в него значение result. В случае, если такой файл уже существует, данные в нем перезаписываются.

fs.appendFileSync('./index.txt', result) - создать файл index.txt и добавить в конец файла значение result. В случае, если такой файл уже существует, данные в нем дополняются.

fs.readDirSync('./node_modules') - посмотреть содержимое папки

fs.statSync('./node_modules') - посмотреть статистику по папке

fs.statFile('./index.txt') - посмотреть статистику по файлу
