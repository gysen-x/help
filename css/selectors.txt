##### КОМБИНАТОРНЫЕ СЕЛЕКТОРЫ #####

Комбинатор "," - позволяет обратиться к нескольким элементам
.child1, .child2 {}

Комбинатор " " - позволяет обратиться ко всем вложенным элементам не зависимо от уровня вложенности
.child1 .child2 {}

Комбинатор ">" - позволяет обратиться к вложенному элементу непосредственного потомка
.child1 > .child2 {}

Комбинатор "" - позволяет обратиться к элементу с комбинацией указанных селекторов 
.child1.child2 {}

Комбинатор "~" - все соседние элементы, позволяет обратиться ко всем элементам, у которых соблюден порядок селекторов: child1 вложен в child2, не зависимо где они находятся
.parent1 .child1 ~ .child2 {}

Комбинатор "+" - следующий элемент, позволяет обратиться ко всем элементам child2, которые следуют за child1, не зависимо где они находятся
.parent1 .child1 + .child2 {}

Селектор по атрибуту "[]" - применяется только к элементам, у которых выполняются условия по атрибутам
a[href] - поиск атрибута
a[href*="search"] - поиск атрибута по указанной строке
a[href^="http://"] - поиск атрибута по началу строки
a[href$=".png"] - поиск атрибута по концу строки


##### КЛЮЧЕВЫЕ СЛОВА #####

display: initial - сбрасывает стили к стилям по умолчанию
display: inherit - наследовать стили от родителей
display: unset - у наследуемых стилей применяет inherit, а у ненаследуемых стилей initial
display: revert - сбрасывает к браузерным стилям по умолчанию


##### ПСЕВДОКЛАССЫ #####

Псевдоклассы состояния:
.child:hover - применять стили при наведении
.child:focus - применять стили при выделении или tab
.child:active - применять стили при нажатии

input[type='radio']:checked - применять стили при активации радиокнопки
input[type='radio']:disabled - применять стили для неактивной кнопки
input[type='email']:valid - применять стили при валидных данных
input[type='email']:invalid - применять стили при невалидных данных

.child:first-child = .child:nth-child(1) - применять стили к первому элементу
.child:last-child = .child:nth-last-child(1) - применять стили к последнему элементу
.child:nth-child(3) - применять стили к третьему элементу
.child:nth-child(2n) - применять стили ко всем четным элементам
.child:nth-child(-n+3) - применять стили к трем первым элементам
.child:not(:nth-child(3))- применять стили ко всем элементам кроме третьего


##### ПСЕВДОЭЛЕМЕНТЫ #####

.child::first-line - применять стили к первой строке элемента
.child::first-letter - применять стили к первой букве элемента
.child::placeholder - применять стили к плэйсхолдеру
.child::selection - применять стили к выделенному тексту

.child::before - (обязательное свойство content='***')
.child::after - (обязательное свойство content='***')


##### ПЕРЕМЕННЫЕ #####
Переменная - область памяти, в которой хранится значение

:root {               - переменная будет работать в глобальной области
--red-color: #fcb;
--text-size: 20px;
}

.element {
color: var(--red-color);
font-size: ver(--text-size);
}


##### COUNTER #####
Замена counter у нумерованного списка

.blocks {
  counter-reset: section;
  list-style: none;
}

.block::after {
  counter-increment: section;
  content: "0" counter(section);
}
